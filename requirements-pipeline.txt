# Data pipeline dependencies -- installed by GitHub Actions workflows
# that run pipeline/*.py scripts (fetching NBM data, generating
# polygons). NOT installed by Railway -- the deployed web app never
# needs any of this, it just serves whatever GeoJSON the pipeline
# already wrote to output/.

# --- Core polygon-generation logic (Track A) ---
numpy
shapely
scikit-image    # raster-to-vector polygonization (see pipeline/polygons.py) -- replaced rasterio, which broke Railway
geojson

# --- Track B: real NBM fetching ---
requests        # used by pipeline/inspect_nbm.py and pipeline/fetch_nbm.py to read NBM's .idx files directly
herbie-data     # handles NBM's AWS/nomads paths + efficient byte-range fetching
pandas          # herbie's inventory() returns a pandas DataFrame
xarray
cfgrib
pyproj          # regrid.py: reprojecting NBM's native grid; polygons.py: true geodesic area calculation
scipy           # regrid.py: griddata interpolation; smoothing.py: neighborhood-max + Gaussian filters
